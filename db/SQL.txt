 CREATE TABLE PROJECT(
   ID INTEGER  PRIMARY KEY  AUTOINCREMENT,
   NAME        TEXT    NOT NULL,
   ORDERNO       INT     NOT NULL,
   ENABLE      BOOLEAN NOT NULL DEFAULT 1,
   BUILDER    TEXT,
   CONTRACTOR  TEXT,
   SUPERVISOR  TEXT,
   DESC        TEXT
);
