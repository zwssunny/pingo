 CREATE TABLE PROJECT(
   ID INT PRIMARY KEY     NOT NULL,
   NAME        TEXT    NOT NULL,
   ORDERNO       INT     NOT NULL,
   BUILDER    TEXT,
   CONTRACTOR  TEXT,
   SUPERVISOR  TEXT,
   DESC        TEXT
);

 CREATE TABLE MENUITEM(
   ID INT PRIMARY KEY     NOT NULL,
   PROJECTID INT NOT NULL,
   NAME        TEXT    NOT NULL,
   ORDERNO       INT     NOT NULL,
   PAGEINDE    INT NOT NULL,
   DESC        TEXT
 );

  CREATE TABLE HIGHLIGHT(
   ID INT PRIMARY KEY     NOT NULL,
   PROJECTID INT NOT NULL,
   NAME        TEXT    NOT NULL,
   ORDERNO       INT     NOT NULL,
   PAGEINDE    INT NOT NULL,
   DESC        TEXT
 );
  CREATE TABLE FEATURES(
   ID INT PRIMARY KEY     NOT NULL,
   PROJECTID INT NOT NULL,
   NAME        TEXT    NOT NULL,
   ORDERNO       INT     NOT NULL,
   DESC        TEXT
 );
